\apisummary{
    This routine tests if an error handler should be called.
}

\begin{apidefinition}

\begin{Csynopsis}
int shmem_errhandler_poll(void);
\end{Csynopsis}

\begin{apiarguments}
\end{apiarguments}

\apidescription{
    This routine verifies if an \openshmem error handler (user-defined or predefined)
    has been invoked at the local \ac{PE}. The error handler may have been invoked
    before, or during the call. When an error is pending (for example, as the result
    of another \ac{PE} invoking the \textit{SHMEM\_ERRHANDLER\_GBREAK} error handler),
    the associated error handler is invoked before the call returns. The operation is
    not synchronizing, that is, a \texit{false} return value does not guarantee that
    no \ac{PE} invoked \textit{SHMEM\_ERRHANDLER\_GBREAK}.
}

\apireturnvalues{
    Returns \textit{true} if an \openshmem error handler has been locally invoked
    (during or before the \textit{shmem_errhandler_poll} call), \textit{false}
    otherwise.
}

\apinotes{
    None.
}

\end{apidefinition}
